var app = angular.module('moduloFiltros', []);
 
app.filter('decorator', function(){
 
	return function(text){
 
		return String(text).toUpperCase();
	}
});
 
app.filter( "removeHtml", function(){
		return function(texto){
			//console.log(texto)
 
			return String(texto).replace(/<[^>]+>/gm, '');
		}
	});
